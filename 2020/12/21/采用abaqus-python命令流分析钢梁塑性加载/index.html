采用abaqus+python命令流分析钢梁塑性加载

二次开发分析 钢梁塑性加载

Mdb()
session.viewports[‘Viewport: 1’].setValues(displayedObject=None)

建立草图

s = mdb.models[‘Model-1’].ConstrainedSketch(name=’profile‘, sheetSize=2.0)
g, v, d, c = s.geometry, s.vertices, s.dimensions, s.constraints
s.setPrimaryObject(option=STANDALONE)
s.rectangle(point1=(0.0, 0.0), point2=(0.5, 0.02))

初始化

p = mdb.models[‘Model-1’].Part(name=’steelbeam’, dimensionality=THREE_D,
type=DEFORMABLE_BODY)
p = mdb.models[‘Model-1’].parts[‘steelbeam’]
p.BaseSolidExtrude(sketch=s, depth=3.2)
s.unsetPrimaryObject()
del mdb.models[‘Model-1’].sketches[‘profile‘]

s1 = mdb.models[‘Model-1’].ConstrainedSketch(name=’profile‘, sheetSize=2.0)
g, v, d, c = s1.geometry, s1.vertices, s1.dimensions, s1.constraints
s1.setPrimaryObject(option=STANDALONE)
s1.CircleByCenterPerimeter(center=(0.0, 0.0), point1=(0.2, 0.0))
p = mdb.models[‘Model-1’].Part(name=’roller’, dimensionality=THREE_D,
type=DISCRETE_RIGID_SURFACE)
p = mdb.models[‘Model-1’].parts[‘roller’]
p.BaseSolidExtrude(sketch=s1, depth=0.5)
s1.unsetPrimaryObject()
